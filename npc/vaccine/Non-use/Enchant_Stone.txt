function	script	25170	{

		end;
		}

.@ref = getequiprefinerycnt( EQI_COSTUME_GARMENT );

.@card2 = getequipcardid( EQI_COSTUME_GARMENT, 1 );

set .@equip_refine, getequiprefinerycnt(EQI_COSTUME_GARMENT);
//set .@enchan1_rand, rand(1,2);
switch(select("인첸트한다:취소")){ 
case 1:
//if(.@enchan1_rand == 1) set .@equip_card[3],29144;
//if(.@enchan1_rand == 2) set .@equip_card[3],29144;
//set .@equip_card[3],29144;
delitem2 .@id, 1,1, .@ref, 0, .@card1, .@card2, .@card3, .@card4;
//delequip EQI_COSTUME_GARMENT;
specialeffect2 EF_REPAIRWEAPON;			
								//getitem2 .@id, 1,1, .@ref, 0, .@card1, .@card2, .@card3, .@card4;
								if ( !.@card4 )
getitem2 .@id, 1,1, .@ref, 0, .@card1, .@card2, .@card3, 25170 + .@o * 10 + .@r;

								close;
								case 2:
								message ""+strcharinfo(0)+"","인첸트가 취소되엇습니다.";
								getitem 50009,1;
								close;
								}
								}



eend:

	end;

enend:
message ""+strcharinfo(0)+"","인첸트가 완료되었습니다.";

}
